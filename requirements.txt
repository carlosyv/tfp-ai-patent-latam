# =============================================================================
# requirements.txt — tfp-ai-patent-latam
# =============================================================================
# Replication package for:
# "The Impact of AI Adoption on Total Factor Productivity (TFP) in Latin America"
#
# The panel estimators (OLS, FE, RE, CCEP, CCEFE, Quantile Regression) are
# implemented from scratch in NumPy — no specialized econometrics package
# (statsmodels / linearmodels) is required. SciPy is used only for general
# numerical routines: normal/t distributions (p-values) and linprog (DEA solver).
#
# Tested on Python 3.10+
# Install with: pip install -r requirements.txt
# =============================================================================

# Core numerical and data libraries
numpy>=1.24.0
pandas>=2.0.0

# General numerical routines: scipy.stats (norm, t) for p-values and
# scipy.optimize.linprog for the DEA linear programs
scipy>=1.10.0

# Required for reading .xlsx input data files (WIPO patent data)
openpyxl>=3.1.0
